DV.prototype.symbol()

문법

var ret = DV.symbol();

파라미터

반환

DVE 객체를 반환합니다.

설명

symobl 메소드는 symbol 요소를 생성합니다.

symbol 요소는 재사용이 가능한 템플릿 형태를 지원하기 위한 컨테이너 요소입니다.

symbol 요소 자체적으로 렌더링되지 않으며, use 요소를 사용하여 표현을 합니다.

symbol 요소는 viewBox, preserveAspectRatio 속성을 가집니다.

symbol 요소를 참조하면 width, height 속성값에 따라 스케일링이 일어 납니다.

<svg width="300" height="100">
  <defs>
    <symbol id="sim1" viewBox="0 0 20 20">
      <rect x="0" y="0" width="9" height="9" fill="red"/>
      <rect x="11" y="0" width="9" height="9" fill="green"/>
      <rect x="0" y="11" width="9" height="9" fill="blue"/>
      <rect x="11" y="11" width="9" height="9" fill="magenta"/>
    </symbol>
  </defs>
  <use xlink:href="#sim1" x="10" y="10" width="20" height="20"/>
  <use xlink:href="#sim1" x="50" y="25" width="50" height="100"/>
</svg>

var svg = DV.svg().size(300,100).into("dv-01");
var def = DV.defs().into(svg);
var sim = DV.symbol().viewbox(0,0,20,20).id("sim1").into(def).append(
    DV.rect().dim(0,0,9,9).fill("red"),
    DV.rect().dim(11,0,9,9).fill("blue"),
    DV.rect().dim(0,11,9,9).fill("green"),
    DV.rect().dim(11,11,9,9).fill("magenta")
);

DV.use().xhref(sim).dim(10,10,20,20).into(svg);
DV.use().xhref(sim).dim(50,25,50,50).into(svg);
