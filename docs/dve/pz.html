DVE.prototype.pz()

문법

var ret = dve.pz();

파라미터

반환

DVE 객체를 반환합니다.

설명

'pz' 메소드는 'path' 요소의 'closepath' 커맨드를 나타냅니다.

closepath는 현재 서브 패스 상에서 현재 위치와 서브 패스의 첫 위치와 선분를 그려, 서브 패스를 닫게합니다.

<svg width="5cm" height="4cm" viewBox="0 0 500 400">
  <path d="M100,100 L400,100 M400,300 L100,300 L100,350 z" fill="none" stroke="blue" stroke-width="5"/>
</svg>

var svg = DV.svg().size("5cm","4cm").viewbox(0,0,500,400).into("dv-01");
DV.path().d("M100,100 L400,100 M400,300 L100,300 L100,350 Z").fill("none").stroke("blue",5).into(svg);

var svg = DV.svg().size("5cm","4cm").viewbox(0,0,500,400).into("dv-02");
DV.path().fill("none").stroke("blue",5)
    .mto(100,100).lto(400,100)
    .mto(400,300).lto(100,300).lto(100,350)
    .pz().into(svg);

var svg = DV.svg().size("5cm","4cm").viewbox(0,0,500,400).into("dv-03");
DV.path().fill("none").stroke("blue",5)
    .mto(100,100).lto(400,100)
    .mto(0, 200,true).lto(-300,0,true).lto(0,50,true)
    .pz().into(svg);
