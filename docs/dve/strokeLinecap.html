DVE.prototype.strokeLinecap()

문법

let ret = object.strokeLinecap(linecap)

파라미터

이름 형식 설명
linecap "butt"|"round"|"square"

테두리 끝을 그리는 방법을 결정합니다.

'butt', 'round', 'square' 값으로 설정할 수 있습니다.

반환

DVE 객체를 반환합니다.

설명

strokeLinecap 메소드는 SVG 요소의 stroke-linecap 속성을 설정합니다.

stroke-linecap 속성은 세그먼트 경로의 양 끝 모양을 결정합니다.

<svg width="350" height="220">
  <path d="M60,60 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#212121" stroke-width="23" stroke-linecap="butt"/>
  <path d="M60,60 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
  <path d="M60,120 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#212121" stroke-width="23" stroke-linecap="round"/>
  <path d="M60,120 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
  <path d="M60,180 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#212121" stroke-width="23" stroke-linecap="square"/>
  <path d="M60,180 l50,-20 150,20 150,-20 150,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
</svg>

let svg = DV.svg().size(350,220).into("dv-01");
DV.path().d("M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121",23,"butt").into(svg);
DV.path().d("M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448aff",3).into(svg);

DV.path().d("M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121",23).attr("stroke-linecap", "round").into(svg);
DV.path().d("M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448aff",3).into(svg);

DV.path().d("M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121",23).strokeLinecap("square").into(svg);
DV.path().d("M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448aff",3).into(svg);
