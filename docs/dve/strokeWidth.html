DVE.prototype.strokeWidth()

문법

var ret = dve.strokeWidth(width)

파라미터

이름 형식 설명
width <length>

테두리의 크기를 나타냅니다. 퍼센티지 형식 및 길이(2, 2px, 2cm, ...)를 나타내는 형식을 사용할 수 있습니다.

퍼센티지 형식을 사용하는 경우, 현재 viewport에 대한 퍼센티지만큼 표현합니다.

0이면 렌더링을 하지 않으며, 음수인 경우 오류가 발생합니다.

반환

DVE 객체를 반환합니다.

설명

strokeWidth 메소드는 SVG 요소의 stroke-width 속성을 설정합니다.

width 파라미터는 테두리의 크기를 설정합니다. stroke 메소드를 사용할 때, width를 명시하지 않았다면 1로 설정됩니다.

storkeWidth 메소드를 사용하기 보다는 stroke 메소드의 width 파라미터를 더 자주 사용하게 됩니다.

<svg width="350" height="220">
  <path d="M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#212121" stroke-width="23"/>
  <path d="M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
  <path d="M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#212121" stroke-width="23"/>
  <path d="M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
  <path d="M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#212121" stroke-width="23"/>
  <path d="M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20" fill="none" stroke="#448AFF" stroke-width="3"/>
</svg>

var svg = DV.svg().size(350,220).into("dv-01");

DV.path().d("M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121",23).into(svg);
DV.path().d("M60,60 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448AFF",3).into(svg);

DV.path().d("M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121").attr("stroke-width",23).into(svg);
DV.path().d("M60,120 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448AFF").attr("stroke-width",3).into(svg);

DV.path().d("M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#212121").strokeWidth(23).into(svg);
DV.path().d("M60,180 l50,-20 l50,20 l50,-20 l50,20 l50,-20").fill("none").stroke("#448AFF").strokeWidth(3).into(svg);
