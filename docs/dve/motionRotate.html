DVE.prototype.motionRotate()

문법

let ret = object.motionRotate(value);

파라미터

이름 형식 설명
value <number>|'auto'|'auto-reverse'
  • <number>: 사용자 좌표계의 원점을 기준으로 주어진 각도(degree)만큼 회전시킵니다.
  • auto: 대상이 이동할 경로의 방향의 각도만큼 시간이 지남에 따라 회전합니다.
  • auto-reverse: 대상이 이동할 경로의 방향의 각도에 180도를 더한 만큼 시간이 지남에 따라 회전합니다.

반환

DVE 객체를 반환합니다.

설명

motionRotate 메소드는 animateMotion 요소의 rotate 속성을 설정할 때 사용합니다.

rotate 속성은 사용자 좌표계의 원점에 대한 회전 변환을 적용하기 위해 애니메이션 대상의 이동 경로와 연산을 할 때 사용됩니다.

이 회전 변환은 path 속성을 통한 경로 계산을 다 마친 후, 추가로 계산됩니다.

<svg width="500" height="200">
    <g>
        <path d="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" stroke-width="1.5" stroke="gray" fill="none"/>
        <path d="M-5,5 L0,-5 L5,5 z" fill="red">
            <animateMotion path="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" begin="0s" dur="5s" rotate="auto" fill="freeze"/>
        </path>
    </g>
    <g transform="translate(0,50)">
        <path d="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" stroke-width="1.5" stroke="gray" fill="none"/>
        <path d="M-5,5 L0,-5 L5,5 z" fill="red">
            <animateMotion path="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" begin="0s" dur="5s" rotate="auto-reverse" fill="freeze"/>
        </path>
    </g>
    <g transform="translate(0,100)">
        <path d="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" stroke-width="1.5" stroke="gray" fill="none"/>
        <path d="M-5,5 L0,-5 L5,5 z" fill="red">
            <animateMotion path="M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0" begin="0s" dur="5s" rotate="45" fill="freeze"/>
        </path>
    </g>
</svg>

크롬 브라우저에서는 auto, auto-reverse만 사용 가능합니다.

let d = "M50,50 c50,-25 50,-25 100,0 s50,25 100,0 s50,-25 100,0 s50,25 100,0";
let svg = DV.svg().size(500,200).into("dv-01");

let g = DV.g().into(svg);
DV.path().d(d).nofill().stroke("gray",1.5).into(g);
DV.path({"d":"M-5,5 L0,-5 L5,5 z"}).fill("red").append(
    DV.animateMotion().motionPath(d).time("0s","5s").motionRotate("auto").freeze()
).into(g);

g = DV.g().translate(0,50).into(svg);
DV.path().d(d).nofill().stroke("gray",1.5).into(g);
DV.path({"d": "M-5,5 L0,-5 L5,5 z"}).fill("red").append(
    DV.animateMotion().motionPath(d).time("0s","5s").motionRotate("auto-reverse").freeze()
).into(g);

g = DV.g().translate(0,100).into(svg);
DV.path().d(d).nofill().stroke("gray",1.5).into(g);
DV.path({"d": "M-5,5 L0,-5 L5,5 z"}).fill("red").append(
    DV.animateMotion().motionPath(d).time("0s","5s").motionRotate(45).freeze()
).into(g);
