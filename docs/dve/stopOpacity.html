dve.prototype.stopOpacity()

문법

var ret = dve.stopOpacity(opacity);

파라미터

이름 형식 설명
opacity <opacity-value>, 일반적으로 0~1

불투명도를 나타냅니다.

반환

dve 객체를 반환합니다.

설명

'stopOpacity' 메소드는 'stop' 요소의 'stop-opacity' 속성을 설정할 때 사용합니다.

'stop' 요소는 gradient를 표현할 때 사용되는 색상 정보를 나타냅니다.

<svg width="600px" height="150px">
    <defs>
        <linearGradient id="S01LG01" x2="50%">
            <stop offset="30%" stop-color="#F60" stop-opacity=".5"/>
            <stop offset="70%" stop-color="#FF6"/>
        </linearGradient>
        <radialGradient id="S01RG01" cx=".25" cy=".25" r="1">
            <stop offset="0%" stop-color="red"/>
            <stop offset="100%" stop-color="blue" stop-opacity=".25"/>
        </radialGradient>
    </defs>
    <rect x="0" y="25" width="200" height="25" fill="url(#S01LG01)"/>
    <rect x="0" y="85" width="200" height="25" fill="url(#S01RG01)"/>
    </g>
</svg>

var svg = DV.svg().size(600,150).into("dv-01");
var def = DV.defs().into(svg);
var grad1 = DV.gradl().point2("50%").id("D01GRL01").into(def);
DV.stop().stopOffset("30%").stopColor("#F60").stopOpacity(.5).into(grad1);
DV.stop().stopOffset("70%").stopColor("#FF6").into(grad1);
var grad2 = DV.gradr().center(.25,.25).radius(1).id("D01GRL02").into(def);
DV.stop().stopOffset("0%").stopColor("red").into(grad2);
DV.stop().stopOffset("100%").stopColor("blue").stopOpacity(.25).into(grad2);

DV.rect().dim(0,25,200,25).fill(grad1).into(svg);
DV.rect().dim(0,85,200,25).fill(grad2).into(svg);
