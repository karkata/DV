DVE.prototype.resettrans()

문법

var ret = dve.resettrans();

파라미터

반환

DVE 객체를 반환합니다.

설명

resettrans 함수는 변환된 좌표계를 원상복귀할 때 사용합니다.

var svg = DV.svg().size(400,150).into("dv-01");
var box = DV.g().into(svg);
DV.rect().dim(10,10,50,50).fill("#ff5722").into(box);

var button = DV.g().css("cursor","pointer").into(svg);
DV.rect().dim(50,100,50,25,8,8).fill("#21212").into(button);
DV.text().txt("Move").xys(75,117).fill("white").tanchor("middle").into(button);
button.bind("click", function (evt) {
    if (box.transformation.length == 0)
        box.translate(100);
    else
        box.resettrans();
    evt.preventDefault();
});
